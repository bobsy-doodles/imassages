tell application "System Events" set macAddress to do shell script "networksetup -getinfo en0 | grep 'Ethernet Address' | awk '{print $3}'" end tell tell application "Mail" set newMessage to make new outgoing message with properties {subject:"MAC Address from " & (system info) & " on " & (current date) & " " & (current time), content:"Please find the MAC address below:" & return & macAddress, visible:true} tell newMessage set sender to "your-email@example.com" -- Replace with your email address set visible to true send end tell end tell
